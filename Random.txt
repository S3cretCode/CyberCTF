        function validateFirst() {
            const input = document.getElementById("bot-check").value.trim();
            const errorMsg1 = document.getElementById("error-1");

            if (input.toLowerCase() === "yes") {
                document.getElementById("captcha-stage-1").classList.add("hidden");
                document.getElementById("captcha-stage-2").classList.remove("hidden");

                document.getElementById("title").classList.add("hidden");
                document.getElementById("meme-img").classList.add("hidden");
                errorMsg1.classList.add("hidden");
            } else {
                errorMsg1.classList.remove("hidden");
            }
        }

        async function validateFinal() {
            const input = document.getElementById("captcha-input").value.trim();
            const errorMsg2 = document.getElementById("error-2");

            const encoder = new TextEncoder();
            const data = encoder.encode(input);
            const hashBuffer = await crypto.subtle.digest("SHA-256", data);
            const hashArray = Array.from(new Uint8Array(hashBuffer));
            const hashHex = hashArray.map(b => b.toString(16).padStart(2, "0")).join("");

            if (hashHex === "7bb7ffc50078c52d99653340b9f3bd50ccc9697bd83f0cfcf399571ca0c81e9a") {
                document.getElementById("captcha-stage-2").classList.add("hidden");
                document.getElementById("final-stage").classList.remove("hidden");
                errorMsg2.classList.add("hidden");
            } else {
                errorMsg2.classList.remove("hidden");
            }
        }
    </script>
